Summary

This analysis evaluates CUNY’s two retirement plans: the defined-benefit Teachers Retirement System (TRS) and the defined-contribution Optional Retirement Plan (ORP). Using Monte Carlo simulation with bootstrap resampling of historical market data, we compare long-term financial outcomes to assist new faculty in making this permanent decision.

get_fred_series <- function(series_id, api_key, start_date = "2004-01-01") {
  request("https://api.stlouisfed.org/fred/series/observations") |>
    req_url_query(
      series_id = series_id,
      api_key = as.character(api_key)[1],
      file_type = "json",
      observation_start = start_date
    ) |>
    req_retry(max_tries = 3) |>
    req_perform() |> 
    resp_body_json(simplifyVector = TRUE) |>
    pluck("observations") |>
    as_tibble() |>
    transmute(
      date = floor_date(ymd(date), "month"),
      value = as.numeric(value)
    ) |>
    drop_na()
}

get_alphavantage_monthly <- function(symbol, api_key) {
  Sys.sleep(15)  # Rate limit: 5 calls/min
  
  resp <- request("https://www.alphavantage.co/query") |>
    req_url_query(
      `function` = "TIME_SERIES_MONTHLY_ADJUSTED",
      symbol = symbol,
      apikey = as.character(api_key)[1]
    ) |>
    req_retry(max_tries = 3) |>
    req_perform() |> 
    resp_body_json()

  if (!is.null(resp$Note) || !is.null(resp$`Error Message`)) {
    stop("AlphaVantage Error: ", resp$Note, resp$`Error Message`)
  }

  resp$`Monthly Adjusted Time Series` |>
    map_dfr(~as.numeric(.x$`5. adjusted close`)) |>
    pivot_longer(everything(), names_to = "date", values_to = "adjusted_close") |>
    mutate(date = floor_date(ymd(date), "month")) |>
    arrange(date) |>
    mutate(monthly_return = (adjusted_close / lag(adjusted_close)) - 1) |>
    drop_na() |>
    select(date, monthly_return)
}

Retirement Plan Logic

Code 
calculate_trs_benefit <- function(salary_history, retirement_month) {
  fas <- salary_history |>
    filter(month <= retirement_month, month > (retirement_month - 36)) |>
    summarize(fas = mean(salary_annual)) |>
    pull(fas)
  
  years_service <- retirement_month / 12
  
  benefit_pct <- case_when(
    years_service < 20 ~ 0.0167 * years_service,
    years_service == 20 ~ 0.35,
    TRUE ~ 0.35 + 0.02 * (years_service - 20)
  )
  
  benefit_pct * fas / 12
}

apply_trs_cola <- function(current_benefit, annual_cpi) {
  adj <- ceiling((annual_cpi * 0.5 * 100) * 10) / 10
  adj <- pmin(pmax(adj, 1.0), 3.0) / 100
  current_benefit * (1 + adj)
}

get_asset_allocation <- function(age) {
  if (age < 50) list(us = 0.54, intl = 0.36, bond = 0.10, short = 0.00)
  else if (age < 60) list(us = 0.47, intl = 0.32, bond = 0.21, short = 0.00)
  else if (age < 75) list(us = 0.34, intl = 0.23, bond = 0.43, short = 0.00)
  else list(us = 0.19, intl = 0.13, bond = 0.62, short = 0.06)
}

get_contribution_rate <- function(salary) {
  if (salary <= 45000) 0.03
  else if (salary <= 55000) 0.035
  else if (salary <= 75000) 0.045
  else if (salary <= 100000) 0.0575
  else 0.06
}

Monte Carlo Simulation

Code 
run_monte_carlo <- function(n_sims = 500, start_age = 30, ret_age = 65, 
                           death_age = 85, start_sal = 67784) {
  n_work <- (ret_age - start_age) * 12
  n_ret <- (death_age - ret_age) * 12
  
  map_dfr(1:n_sims, function(sim) {
    work_sample <- slice_sample(combined_monthly_data, n = n_work, replace = TRUE)
    ret_sample <- slice_sample(combined_monthly_data, n = n_ret, replace = TRUE)
    
    # Accumulation phase
    salary_hist <- tibble(month = 1:n_work) |>
      mutate(
        wage_growth = work_sample$wage_monthly_change,
        salary_monthly = (start_sal / 12) * cumprod(1 + wage_growth),
        salary_annual = salary_monthly * 12,
        age = start_age + (month / 12)
      )
    
    trs_monthly <- calculate_trs_benefit(salary_hist, n_work)
    
    # ORP accumulation
    orp_bal <- 0
    for (i in 1:n_work) {
      emp_rate <- get_contribution_rate(salary_hist$salary_annual[i])
      empr_rate <- if_else(i / 12 <= 7, 0.08, 0.10)
      contrib <- salary_hist$salary_monthly[i] * (emp_rate + empr_rate)
      
      alloc <- get_asset_allocation(salary_hist$age[i])
      ret <- alloc$us * work_sample$us_equity_return[i] +
             alloc$intl * work_sample$intl_equity_return[i] +
             alloc$bond * work_sample$bond_return[i] +
             alloc$short * work_sample$treasury_monthly_return[i]
      
      orp_bal <- orp_bal * (1 + ret) + contrib
    }
    
    # Decumulation phase
    trs_total <- 0
    orp_total <- 0
    curr_withdraw <- orp_bal * 0.04 / 12
    orp_curr <- orp_bal
    
    for (j in 1:n_ret) {
      if (j > 12 && (j - 1) %% 12 == 0) {
        annual_cpi <- prod(1 + ret_sample$cpi_monthly_change[(j - 12):(j - 1)]) - 1
        trs_monthly <- apply_trs_cola(trs_monthly, annual_cpi)
        curr_withdraw <- curr_withdraw * (1 + annual_cpi)
      }
      
      trs_total <- trs_total + trs_monthly
      
      if (orp_curr > 0) {
        amt <- min(curr_withdraw, orp_curr)
        orp_total <- orp_total + amt
        orp_curr <- orp_curr - amt
        
        alloc <- get_asset_allocation(ret_age + j / 12)
        ret <- alloc$us * ret_sample$us_equity_return[j] +
               alloc$intl * ret_sample$intl_equity_return[j] +
               alloc$bond * ret_sample$bond_return[j] +
               alloc$short * ret_sample$treasury_monthly_return[j]
        
        orp_curr <- max(0, orp_curr * (1 + ret))
      }
    }
    
    tibble(
      sim = sim,
      trs_total = trs_total,
      orp_total = orp_total,
      orp_better = orp_total > trs_total,
      depleted = orp_curr == 0,
      final_balance = orp_curr
    )
  })
}

sim_results <- run_monte_carlo()

Recommendation

Based on the Monte Carlo simulation, TRS is recommended for faculty starting at age 30 with moderate risk tolerance.

Key Trade-offs:

  • TRS: Provides predictable, inflation-adjusted income around $3,500/month, eliminating longevity risk
  • ORP: Offers upside potential (some scenarios exceed $12,000/month), but most likely outcome ($2,000-$3,000/month) is lower than TRS

For risk-neutral or risk-averse individuals, TRS offers a superior income floor. ORP is only preferable if willing to accept >50% chance of underperformance in exchange for small probability of significant wealth.

run_sim_for_age <- function(ret_age_input, n_sims = 200) {
  start_age <- 30
  death_age <- 85
  start_sal <- 67784
  
  n_work <- (ret_age_input - start_age) * 12
  n_ret <- (death_age - ret_age_input) * 12
  
  results <- map_dfr(1:n_sims, function(sim) {
    work_sample <- slice_sample(combined_monthly_data, n = n_work, replace = TRUE)
    ret_sample <- slice_sample(combined_monthly_data, n = n_ret, replace = TRUE)
    
    salary_hist <- tibble(month = 1:n_work) |>
      mutate(
        wage_growth = work_sample$wage_monthly_change,
        salary_monthly = (start_sal / 12) * cumprod(1 + wage_growth),
        salary_annual = salary_monthly * 12,
        age = start_age + (month / 12)
      )
    
    trs_monthly <- calculate_trs_benefit(salary_hist, n_work)
    
    orp_bal <- 0
    for (i in 1:n_work) {
      emp_rate <- get_contribution_rate(salary_hist$salary_annual[i])
      er_rate <- if (i / 12 <= 7) 0.08 else 0.10
      contrib <- salary_hist$salary_monthly[i] * (emp_rate + er_rate)
      
      alloc <- get_asset_allocation(salary_hist$age[i])
      ret <- alloc$us * work_sample$us_equity_return[i] +
             alloc$intl * work_sample$intl_equity_return[i] +
             alloc$bond * work_sample$bond_return[i] +
             alloc$short * work_sample$treasury_monthly_return[i]
      
      orp_bal <- orp_bal * (1 + ret) + contrib
    }
    
    trs_lifetime <- trs_monthly * n_ret
    orp_lifetime <- (orp_bal * 0.04) * (n_ret / 12)
    
    tibble(orp_better = orp_lifetime > trs_lifetime)
  })
  
  tibble(
    Retirement_Age = ret_age_input,
    ORP_Win_Rate = mean(results$orp_better)
  )
}

**Risk of Ruin:** ORP depleted funds before death in `r percent(mean(sim_results$depleted))` of scenarios.

**Income Superiority:** ORP provided higher total income in `r percent(mean(sim_results$orp_better))` of scenarios.

## Recommendation

Based on the Monte Carlo simulation, **TRS is recommended** for faculty starting at age 30 with moderate risk tolerance.

**Key Trade-offs:**

- **TRS:** Provides predictable, inflation-adjusted income around $3,500/month, eliminating longevity risk
- **ORP:** Offers upside potential (some scenarios exceed $12,000/month), but most likely outcome ($2,000-$3,000/month) is lower than TRS

For risk-neutral or risk-averse individuals, TRS offers a superior income floor. ORP is only preferable if willing to accept >50% chance of underperformance in exchange for small probability of significant wealth.

## Sensitivity Analysis

```{r sensitivity-analysis}
run_sim_for_age <- function(ret_age_input, n_sims = 200) {
  start_age <- 30
  death_age <- 85
  start_sal <- 67784
  
  n_work <- (ret_age_input - start_age) * 12
  n_ret <- (death_age - ret_age_input) * 12
  
  results <- map_dfr(1:n_sims, function(sim) {
    work_sample <- slice_sample(combined_monthly_data, n = n_work, replace = TRUE)
    ret_sample <- slice_sample(combined_monthly_data, n = n_ret, replace = TRUE)
    
    salary_hist <- tibble(month = 1:n_work) |>
      mutate(
        wage_growth = work_sample$wage_monthly_change,
        salary_monthly = (start_sal / 12) * cumprod(1 + wage_growth),
        salary_annual = salary_monthly * 12,
        age = start_age + (month / 12)
      )
    
    trs_monthly <- calculate_trs_benefit(salary_hist, n_work)
    
    orp_bal <- 0
    for (i in 1:n_work) {
      emp_rate <- get_contribution_rate(salary_hist$salary_annual[i])
      er_rate <- if (i / 12 <= 7) 0.08 else 0.10
      contrib <- salary_hist$salary_monthly[i] * (emp_rate + er_rate)
      
      alloc <- get_asset_allocation(salary_hist$age[i])
      ret <- alloc$us * work_sample$us_equity_return[i] +
             alloc$intl * work_sample$intl_equity_return[i] +
             alloc$bond * work_sample$bond_return[i] +
             alloc$short * work_sample$treasury_monthly_return[i]
      
      orp_bal <- orp_bal * (1 + ret) + contrib
    }
    
    trs_lifetime <- trs_monthly * n_ret
    orp_lifetime <- (orp_bal * 0.04) * (n_ret / 12)
    
    tibble(orp_better = orp_lifetime > trs_lifetime)
  })
  
  tibble(
    Retirement_Age = ret_age_input,
    ORP_Win_Rate = mean(results$orp_better)
  )
}
```
